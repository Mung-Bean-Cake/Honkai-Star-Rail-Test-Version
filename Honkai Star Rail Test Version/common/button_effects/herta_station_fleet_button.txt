herta_fleet_district_1_show_effect = {
    potential = {
        always = yes
    }
    allow = {
        always = yes
    }
    effect = {
        custom_tooltip = herta_fleet_district_1_show_effect_text
    }
}
herta_fleet_district_1_build_effect = {
    potential = {
        always = yes
    }
    allow = {
        custom_tooltip = {
            fail_text = need_less_limit
            hidden_trigger = {
                event_target:herta_fleet = {
                    check_variable = {
                        which = herta_station_district_1_num
                        value <= event_target:herta_fleet.herta_station_district_1_num_max
                    }
                    check_variable = {
                        which = herta_station_district_1_num_max
                        value < 10
                    }
                }
            }
        }
        from = {
            resource_stockpile_compare = { resource = energy value >= 1000 }
            resource_stockpile_compare = { resource = minerals value >= 1000 }
            resource_stockpile_compare = { resource = alloys value >= 500 }
        }
    }
    effect = {
        custom_tooltip = herta_fleet_district_1_build_effect_txt
        hidden_effect = {
            event_target:herta_fleet = {
                change_variable = {
                    which = herta_station_district_1_num_max
                    value = 1
                }
                change_variable = {
                    which = herta_station_district_1_num
                    value = 1
                }
            }
            from = {
                add_resource = {
                    energy = -1000
                    minerals = -1000
                    alloys = -500
                }
            }
        }
    }
}

herta_station_district_1_num_count_sub_effect = {
    potential = {
        always = yes
    }
    allow = {
        custom_tooltip = {
            fail_text = need_more_zero
            hidden_trigger = {
                event_target:herta_fleet = {
                    check_variable = {
                        which = herta_station_district_1_num
                        value > 0
                    }
                }
            }
        }
    }
    effect = {
        hidden_effect = {
            event_target:herta_fleet = {
                change_variable = {
                    which = herta_station_district_1_num
                    value = -1
                }
                if = {
                    limit = {
                        check_variable = {
                            which = herta_station_district_1_workforce
                            value > value:herta_station_district_1_pop_max
                        }
                    }
                    set_variable = {
                        which = herta_station_district_1_workforce
                        value = value:herta_station_district_1_pop_max
                    }
                }
            }
        }
    }
}
