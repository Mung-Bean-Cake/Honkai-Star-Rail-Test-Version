simulated_cosmos_flag_change_effect = {
    allow = {
        always = yes
    }
    potential = {
        always = yes
    }
    effect = {
        custom_tooltip = simulated_cosmos_flag_change_effect
        from = {
            hidden_effect = {
                change_variable = {
                    which = simulated_cosmos_flag
                    value = 1
                }
                if = {
                    limit = {
                        check_variable = {
                            which = simulated_cosmos_flag
                            value >=5
                        }
                    }
                    set_variable = {
                        which = simulated_cosmos_flag
                        value = 0
                    }
                }
            }
        }
    }
}
