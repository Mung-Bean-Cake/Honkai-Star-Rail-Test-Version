    }

    }
    effect = {

    }

hsr_path_cunhu_set_primary_effect = {
    allow = {
        custom_tooltip = hsr_path_cunhu_set_primary_effect
        from = {resource_stockpile_compare = { resource = unity value >= 10000 }}
        custom_tooltip = {
            fail_text = hsr_cunhu_primary_tt
            hidden_trigger = {
                from = {
                    not = {
                        has_country_flag = hsr_cunhu_primary
                    }
                }
            }
        }
        custom_tooltip = {
            fail_text = hsr_cunhu_secondary_tt
            hidden_trigger = {
                from = {
                    not = {
                        has_country_flag = hsr_cunhu_secondary
                    }
                }
            }
        }
        custom_tooltip = {
            fail_text = hsr_cunhu_has_tt
            hidden_trigger = {
                from = {
                    has_tradition = tr_cunhu
                }
            }
        }
        custom_tooltip = {
            fail_text = hsr_cunhu_primary_need_tt
            hidden_trigger = {
                from = {
                    check_variable_arithmetic = {
                        which = value:country_path_cunhu_value
                        value > 80
                    }
                }
            }
        }
        custom_tooltip = {
            fail_text = hsr_cunhu_primary_count_need_tt
            hidden_trigger = {
                from = {
                    check_variable_arithmetic = {
                        which = path_primary_count
                        value < path_primary_count_max
                    }
                }
            }
        }
    }
    potential = {
        always = yes
    }
    effect = {
        custom_tooltip = hsr_path_cunhu_set_primary_effect
        hidden_effect = {
            from = {
                set_country_flag = hsr_cunhu_primary
                change_variable = {
                    which = path_primary_count
                    value = 1
                }
                add_resource = { unity = -10000 }
            }
        }
    }
}
