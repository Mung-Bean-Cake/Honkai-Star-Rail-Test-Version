xianzhou_district_1_build_effect = {
    allow = {
        always = yes
    }
    potential = {
        always = yes
    }
    effect = {
        hidden_effect = {
            if = {
                limit = {
                    not = { has_fleet_flag = xianzhou_district_1_build_ongoing }
                }
                set_fleet_flag = xianzhou_district_1_build_ongoing
                set_variable = {
                    which = xianzhou_district_1_build_progress
                    value = this.value:xianzhou_district_build_time
                }
                change_variable = {
                    which = xianzhou_district_1_build_sequence
                    value = 1
                }
                fleet_event = {
                    id = xianzhou.30
                    days = 1
                }
            }
            else_if = {
                limit = {
                    has_fleet_flag = xianzhou_district_1_build_ongoing
                }
                change_variable = {
                    which = xianzhou_district_1_build_sequence
                    value = 1
                }
            }
        }
    }
}
