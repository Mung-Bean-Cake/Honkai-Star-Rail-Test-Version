    optimize_memory
    if = {
        limit = { has_specimen = $KEY_NAME$ }
        change_variable = {
            which = yuzhousuipian
            value = $VALUE$
        }
    }
    else = {
        give_specimen = { key = $KEY_NAME$ origin = $KEY_ORIGIN$ }
    }
}

}
