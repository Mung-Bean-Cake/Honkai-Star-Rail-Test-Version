                    which = xianzhou_district_$district_num$_build_progress

                    which = xianzhou_district_$district_num$_build_sequence
                    value = 1
                }
                fleet_event = {
                    id = $build_sequence_event$
                    days = 1
                }
            }
            else_if = {
                limit = {
                    has_fleet_flag = xianzhou_district_$district_num$_build_ongoing
                }
                change_variable = {
                    which = xianzhou_district_$district_num$_build_sequence
                    value = 1
                }
            }
            from = {

                    alloys = -1

            }
        }
    }
