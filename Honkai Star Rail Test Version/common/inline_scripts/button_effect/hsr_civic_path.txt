hsr_star_civic_$value$_detial_effect = {
    allow = {
        always = yes
    }
    potential = {
        always = yes
    }
    effect = {
        custom_tooltip = hsr_star_civic_$value$_detial_effect
    }
}
hsr_star_civic_$value$_start_effect = {
    allow = {
        custom_tooltip = {
            fail_text = you_have_selected
            hidden_trigger = {
                from = {
                    not = {
                        is_path_civic = yes
                    }
                }
            }
        }
        custom_tooltip = {
            fail_text = need_selected_slot
            hidden_trigger = {
                from = {
                    is_path_slot = yes
                }
            }
        }
        custom_tooltip = {
            fail_text = need_$primary$_primary
            hidden_trigger = {
                from = {
                    has_country_flag = hsr_$primary$_primary
                }
            }
        }
        custom_tooltip = {
            fail_text = need_$secondary$_secondary
            hidden_trigger = {
                from = {
                    has_country_flag = hsr_$secondary$_secondary
                }
            }
        }
    }
    potential = {
        always = yes
    }
    effect = {
        custom_tooltip = start_civic
        hidden_effect = {
            from = {
                hsr_civic_select = {civic_name = $civic_name$  value = $value$ }
            }
        }
    }
}
hsr_star_civic_$value$_reset_effect = {
    allow = {
        custom_tooltip = {
            fail_text = need_selected
            hidden_trigger = {
                from = {
                    has_country_flag = $civic_name$
                }
            }
        }
        custom_tooltip = {
            fail_text = need_selected_slot
            hidden_trigger = {
                from = {
                    is_path_slot = yes
                }
            }
        }
    }
    potential = {
        always = yes
    }
    effect = {
        custom_tooltip = reset_civic
        hidden_effect = {
            from = {
                hsr_civic_reset = {civic_name = $civic_name$}
            }
        }
    }
}
