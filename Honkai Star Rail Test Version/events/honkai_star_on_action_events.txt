country_event = {
    id = hsr_on.1001
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        FROMFROM = {
            exists = owner
            owner = {
                is_ai = no
                any_owned_leader = {
                    is_councilor = yes
                    has_trait = leader_daherta_t3
                }
            }
        }
    }
    immediate = {
        if = {
            limit = {
                has_country_resource = {
                    type = alloys
                    amount >= 90
                }
                has_country_resource = {
                    type = influence
                    amount >= 45
                }
                NOT = {
                    from = {
                        any_system = {
                            distance = {
                                source = from
                                min_jumps = 1
                                max_jumps = 1
                                type = hyperlane
                                use_bypasses = no
                            }
                            exists = owner
                            owner = {
                                OR = {
                                    is_country_type = fallen_empire
                                    is_country_type = awakened_fallen_empire
                                }
                                has_ethic = ethic_fanatic_xenophobe
                            }
                        }
                    }
                }
            }
            add_resource = { alloys = -90 }
            add_resource = { influence = -45 }
            from = {
                create_starbase = {
                    size = starbase_outpost
                    owner = root
                }
                every_system_planet = {
                    if = {
                        limit = {
                            has_deposit_for = shipclass_mining_station
                            has_mining_station = no
                        }
                        create_mining_station = {
                            owner = root
                        }
                    }
                    if = {
                        limit = {
                            has_deposit_for = shipclass_research_station
                            has_research_station = no
                        }
                        create_research_station = {
                            owner = root
                        }
                    }
                }
            }
        }
    }
}

country_event = {
    id = hsr_on.10001
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        has_country_flag = hsr_monthly_join_us
    }

    immediate = {
        if = {
            limit = { any_owned_species = {
                NOR = {
                    is_same_species = ROOT.owner_main_species
                    has_trait = trait_herta_robot_pop_1
                }
            }}
            # 限制每天只改造一个人口
            if = {
                limit = { NOT = { has_country_flag = hsr_monthly_us_enabled }}
                random_owned_pop_group = {
                    limit = {
                        NOR = {
                            is_same_species = ROOT.owner_main_species
                            has_trait = trait_herta_robot_pop_1
                        }
                    }
                    planet = { create_pop_group = { size = prev.trigger:pop_group_size species = ROOT.owner_main_species }}
                    kill_pop_group = { pop_group = this percentage = 1 }
                }
                set_timed_country_flag = { flag = hsr_monthly_us_enabled days = 1 }
                country_event = { id = hsr_on.10001 days = 1 }
            } else = {
                remove_country_flag = hsr_monthly_us_enabled
            }
        }
    }
}
