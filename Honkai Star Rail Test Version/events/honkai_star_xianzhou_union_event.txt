fleet_event = {
    id = xianzhou.30
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        if = {
            limit = {
                has_fleet_flag = xianzhou_district_1_build_ongoing
                check_variable = {
                    which = xianzhou_district_1_build_sequence
                    value > 0
                }
            }
            change_variable = {
                which = xianzhou_district_1_build_progress
                value = -1
            }
            if = {
                limit = {
                    check_variable = {
                        which = xianzhou_district_1_build_progress
                        value <= 0
                    }
                }
                change_variable = {
                    which = xianzhou_district_1_build_sequence
                    value = -1
                }
                change_variable = {
                    which = xianzhou_district_1_num
                    value = 1
                }
                set_variable = {
                    which = xianzhou_district_1_build_progress
                    value = this.value:xianzhou_district_build_time
                }
            }
            fleet_event = {
                id = xianzhou.30
                days = 1
            }
        }
        else = {
            remove_fleet_flag = xianzhou_district_1_build_ongoing
        }
    }
}
