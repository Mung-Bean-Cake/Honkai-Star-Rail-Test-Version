planet_event = {
	id = Astral_Express.02
	title = Death_of_Akivili.500.name
	desc = Death_of_Akivili.500.desc
	picture = GFX_evt_robot_assembly
	show_sound = event_ship_landing
	location = this
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		limit = { has_planet_flag = Create_Express_Welt }
		create_colony = {
			owner = solar_system.owner
			species = solar_system.owner.owner_main_species
		}		
		while = {
			count = 8
			create_pop_group = {
				size = 100
				species = owner_main_species
			}
			remove_district = district_city
			set_name = Express_Welt
		}
		while = {
			count = 10
			remove_district = district_city
		}
	}
}
