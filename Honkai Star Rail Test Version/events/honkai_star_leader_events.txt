    title = "hsr_paragon.63.name"
    desc = "tuopa_catch_phrase"
    is_triggered_only = yes

    picture = GFX_leader_recruitment_bg_legendary_beholder
    event_window_type = leader_recruit

    picture_event_data = {
        portrait = event_target:tuopa_leader
        room = paragon_keides_room
    }
    immediate = {
        hidden_effect = {
            SR_leader_creator = {
                SPECIES_NAME = star_rail_leader
                SPECIES_CLASS = Star_Rail
                LEADER_PORTRAIT = tuopa
                SKILL = 1
                CLASS = scientist
                NAME = NAME_tuopa
                GENDER = female
                HIDEAGE = yes
                ETHIC = ethic_egalitarian
                CATCH_PHRASE = tuopa_catch_phrase
                DESC = tuopa_backstory
                EVENT_TARGET = tuopa_leader
            }
            last_created_leader = {
                remove_all_positive_traits = yes
                add_trait = {trait = leader_tuopa_t1 show_message = no}
                add_trait = {trait = leader_tuopa_t2 show_message = no}
                add_trait = {trait = leader_tuopa_t3 show_message = no}
                change_background_job = researcher
            }
        }
    }
    option = {
        trigger = { }
        name = hsr_paragon.63.a
        is_dialog_only = yes
        response_text = hsr_paragon.63.a.response
    }
    option = {
        trigger = { }
        name = hsr_paragon.63.b
        is_dialog_only = yes
        response_text = hsr_paragon.63.b.response
    }
    option = {
        name = hsr_paragon.63.c
        response_text = hsr_paragon.63.c.response
        is_dialog_only = yes
    }
    option = {
        name = HIRE
        tag = hire_leader
        hidden_effect = {
            SR_hire_leader = {
                EVENT_TARGET = tuopa_leader
            }
            set_country_flag = tuopa_set
            last_created_leader = {
                set_leader_flag = legendary_leader
                save_global_event_target_as = tuopa_leader
                set_leader_flag = immune_to_negative_traits
                set_leader_flag = tuopa
            }
        }
    }
    option = {
        # DISMISS
        name = DISMISS
        tag = dismiss_leader
        hidden_effect = {
            event_target:tuopa_leader = {
                kill_leader = { show_notification = no }
            }
        }
    }
    after = {hidden_effect = {country_event = { id = hsr_ui.63}}}
}
